# ADK 概念示意（API 随版本变，理解结构即可）
from google.adk import Agent

agent = Agent(
    name="ops_agent",
    model="gemini-2.0-flash",
    instruction="你是运维助手",
    tools=[query_disk_tool],
)
response = agent.run("查一下磁盘")
