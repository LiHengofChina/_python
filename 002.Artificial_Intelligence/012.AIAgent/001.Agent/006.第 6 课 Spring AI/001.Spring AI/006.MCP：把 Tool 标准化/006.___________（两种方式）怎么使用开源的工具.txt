spring:
  ai:
    mcp:
      client:
		//============= （第一种配置方法）连已经跑起来的 MCP 服务（你要先启动 8104，Client 再去连）
        sse:
          connections:
            ops-server:
              url: http://127.0.0.1:8104       # 你的
            weather-server:
              url: http://localhost:8080       # 开源/别人的 HTTP MCP #远程已启动的服务（HTTP），如你的 8104

		//============= （第二种配置方法）自己启动时带一个MCP起来， Client 启动时用 npx 拉起一个 Node 进程跑 filesystem MCP。 需 Node.js + npx
        stdio:
          connections:
            filesystem:
              command: npx
              args:
                - -y
                - "@modelcontextprotocol/server-filesystem"
                - "E:\\_python"
